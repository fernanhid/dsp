def BiasPmf(pmf, label):
    new_pmf = pmf.Copy(label=label)

    for x, p in pmf.Items():
        new_pmf.Mult(x, x)
        
    new_pmf.Normalize()
    
    return new_pmf
  
    
pmf_bias = BiasPmf(pmf, 'biased')
