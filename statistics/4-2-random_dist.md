random_list = []
for i in range(1000):
    random_list.append(random.random())

random_cdf = thinkstats2.Cdf(random_list)

thinkplot.PrePlot(2)
thinkplot.Pmfs([random_cdf, random_pmf])
thinkplot.Show()
